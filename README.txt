main.py [commands]

Commands
--fname
	the location of the image that is to be aligned
--auto 
	automatically calculate the depth of pyramids and the search window size
--diff [num]
	search window size
--levels [num]
	depth of pyramids
--contrast
	automatically set the contrast
--crop
	automatically crop the borders of the image
--align_metric [ncc/ssd]
	choose the aligning metric
--all
	align all the images in the ./data folder; output the images in ./out_path
